import createSearch from './headersearch';
import createNotification from './headernotification';
import createMessageIcon from './headermessage';
import createProfileIcon from './headericonprofile';
import createProfileDropdown from './headerprofiledropdown';

document.addEventListener("DOMContentLoaded", function () {
    // Create the header element
    const header = document.createElement("header");
    header.className = "navbar navbar-expand-lg navbar-light bg-light border-bottom py-2";

    // Create a container for the header content
    const container = document.createElement("div");
    container.className = "container-fluid d-flex align-items-center flex-wrap";

    // Add Search Bar
    const searchBar = createSearch();
    container.appendChild(searchBar);

    // Add Notification Bell
    const notificationBell = createNotification();
    container.appendChild(notificationBell);

    // Add Message Icon
    const messageIcon = createMessageIcon();
    container.appendChild(messageIcon);


    // Add Profile Dropdown
    const profileDropdown = createProfileDropdown();
    container.appendChild(profileDropdown);

    // Append the container to the header
    header.appendChild(container);

    // Add Blue Line
    const blueLine = document.createElement("div");
    blueLine.style.backgroundColor = "#007BFF"; // Bootstrap primary blue color
    blueLine.style.height = "4px";
    blueLine.style.width = "100%";

    // Add header and blue line to the DOM
    document.body.prepend(blueLine);
    document.body.prepend(header);

    
});
